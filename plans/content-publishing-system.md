class RayStateSiteSpec {
public:
  // Type of ray state to which this site specification is to apply
  enum class SiteSpecType {
    None = 0,                 // No site specification; all states in search space
    RayStateFarPointHit,     // Farmost point inserted as recoil hit
    RayStateFarPoint,        // Farmost point on line
    RayStateOnSpherePoint,   // Closest point on sphere:
                             // (1) radius = 0: then apply search.sphere and search.tanDist
                             // (2) radius > 0: then search along ray and sphere
    RayStateTangentToCapsule,  // Far and near tangent points to the nearest geometric capsule
    RayStateTangentToSphere,   // Far and near tangent points to the nearest sphere
    // Far exchange between two sphere
    RayStateSphereMustEngineered, // Far outside a knowledge base based "engineered" sphere
  };

  /* The site spec type determines to which states in an $R state space the site representative, a few representative rays, 
     and a radius of tangency will be applied. */
  SiteSpecType specificType = SiteSpecType.None;
  // If (true), it is assumed that the radii of the nearest geometric sphere is smaller than the summation of the two expected radii
  // of the best far ray and the best near ray. When running the candidate heuristic engine, this is guaranteed in a large number
  // of cases. Otherwise, the information of near tangent points of two spheres will be considered and two tangency radii will be provided.
  // If (false), the distance between two centers of the spheres is larger than the summation of the two expected radii of the far and near rays.
  // In this case, neither the information of the near tangent points of two spheres nor the information of two tangency radii will be stored, but instead,
  // a default $curLayerTrunkVector will be constructed and only the far tangent points of the two spheres will be kept.
  bool hasTangentPointInformation = false;
  bool hasTangentRadius = false; // Whether or not to show the tangency radii information with respect to two centers
  bool hasFineNearSTATE = false;  // Whether or not the exact tangent point of near state is supplied
  bool hasSurfaceNormal = false;
  bool onlyIsActive = false; // Only returned if true
  bool onlyIsNear = false; // Only returned if true; if (true) pointTrajectoryRadius can contain an associated :$\rho$
  bool includeFarState = false; // Whether or not to show the far most ray against the sphere
  bool includeNearState = false; // Whether or not to show the near most ray against the sphere
  bool includeNearSphereState = false; // Whether or not to show the near most ray against the sphere
  bool includeNearBallState = false; // Whether or not to show the near most ray against the sphere
  bool allowSmallSearchRegion = false;
  double additionalSphereRadiusInSecantMethod = 0;

  // Given one point on a ray and the line of the ray, return the
  // nearest point-apex on the sphere described by the given dimension and the tip.
  // This is the sphere described in a previous version of the algorithm.
  //  \vector{p}: The normal vector to this sphere.
  //  \pi : The dot product of the normal vector and a point on the sphere.
  //  $state: The point on the ray.
  //  $t : If the false is returned, the value of $t is meaningless.
  //  $perpFactor : If the false is returned, the value of $perpFactor is meaningless.
  //  \return true if the intersection is outside an associated search space with the hyperplane.
  //  \white{\begin{algorithmic}
  //        \UsingParameter{\vector{p}, \pi, $\state$}
  //        \UsingTools{Compare}
  //        \State \vector{p' in $\pi$ (of $f_pigger$)}
  //        \If {false}
  //            \Return false
  //        \EndIf
  //   linear Ax^2 + Bx + C = 0, epsilon = 1e-12; evaluation by evaluating the first order of $A$
  //        \For{$i = 0$ to $n$}
  //            \If {Ax^2 + Bx + C divide by little-epsilon?}
  //                \If < $epsilon}
  //                    \Return false
  //                \EndIf
  //            \EndIf
  //        \EndFor
  //        \Return true
  //  \end{algorithmic}}
  // Modified from [Beintema, ""]{CLEEN 1997}
  bool getCircleSphereInnerApex(
      const VectorPP &vSphereRadius,
      const VectorPP &vSphereTip,
      const RayState &state,
      double *tOuter,
      double *perpFactorOuter);

  // Given one point \vector{d} on a ray and the line of the ray, return the
  // nearest point-apex on the sphere described by the given dimension and the tip.
  //  $vSphereRadius: The normal vector to this sphere.
  //  $vSphereTip: The dot product of the normal vector and a point on the sphere.
  //  $state : The point on the ray.
  //  $perpFactor: Perpendicular distance to the apex of the section. $\pi / 4 < perpFactor <= \pi / 2$.
  //  $pointTrajectoryRadius: First order interpolation between points on a kernel curve. In many cases, $pointTrajectoryRadius = t * vSphereRadius.
  //  The return value \perpFactorOuter is meaningful only if the returned value is true. If the returned value is false, the value of $tOuter is meaningless.
  //  \white{\begin{algorithmic}
  // \UsingPattern{Switch}
  // \State \vector{d} in $\sum\limits_{n\in\mathbb{N}}$ \vector{d_n}
  // \State \vector{d'} in $\sum\limits_{k=1}^{n} \frac{d_k}{|d_k|}$
  // \State \vector{d'\perp} = \vector{d'} - $\frac{d'}{|d'|}$
  // \State \vector{d'\perp} = \vector{d'\perp} - $\frac{\vector{d'} \cdot \vector{d'\perp}}{|\vector{d'\perp}|} \frac{\vector{d'\perp}}{|\vector{d'\perp}|}$
  // \State pointTrajectoryRadius = \vector{d}\perp (prepend $\vector{d}$ to pointTrajectoryRadius)
  // \State $i_backward_n2ilus = 0$
  // \State $i_forward_n2ilus = n-1$
  // \For{$i = 0$ to $n-1$}
  // \If \vector{d}\perp = $\frac{d}{|d|}$
  // \State $i_backward_n2ilus = i$
  // \ElseIf \vector{d}\perp = $-\frac{d}{|d|}$
  // \State $i_forward_n2ilus = i$
  // \EndIf
  // \EndFor
  // \State $pointTrajectoryRadius.append( \vector{d'_top} - i_backward_n2ilus\cdot vSphereRadius )$
  // \State $pointTrajectoryRadius.append( \vector{d'_top} - i_forward_n2ilus\cdot vSphereRadius)$
  // \State \vector{d'_top} =  $i_backward_n2ilus\cdot vSphereRadius + i_forward_n2ilus\cdot vSphereRadius$
  // \Statex
  // \White{$m_p(...) \gets \vector{d}_p(...) \cdot \vector{d}_p(...) + \vector{d}_p(...) \cdot \vector{d}_p(...) vSphereRadius \cdot \vector{d}_p(...)}$  for linear and direction evaluation
  // \White{\begin{align*} 
  // m_p(...) &= \vectd{\vector{d',k} p} \cdot \vectd{\vector{d',k} p} \cdot (\vector{d',k} p) + vSphereRadius \cdot \vector{d'_k} \cdot \vector{d'_k} \\
  //          &= \vectd{\vector{d',k} p} \cdot \vector{"} k} \cdot \dfrac{\vectd{\vector{d',k} p} \cdot \vector{d',k} p(\cdot\vectd\vector{d',k} k}, k)^2 \cdot \frac{1}{\vectd\vector{d',k} k} \log(\vl{\vector{d',k} p}) + 2 vSphereRadius \cdot \vector{d'_k} \dfrac{\dscal{d,k} 0}{\dscal{d,k} 1} \dfrac{\dscal{d,k} 0}{\dscal{d,k} 1} + vSphereRadius \cdot \vector{d'_k} \dfrac{\ddddot{vSphereRadius}{k} 0}{\ddddot{vSphereRadius}{k} 2} \dfrac{\ddddot{vSphereRadius}{k} 0} {\ddddot{vSphereRadius}{k} 2}.

def divide_parrays_into_chunks(parrays, n_chunks, n_threads, chunk_size):
    """
    Divide a parray into chunks for multiprocessing. Workload is distributed
    by distributing the parrays over the number of workers.
    Returns:
        List of chunks, each chunk contains the worker having indices corresponding
        to the workers having the same indices across all the workers.
    """
    parray_chunks = parrays.map(lambda x: x).collect().getSaveRate
    parray_chunks[0] /= n_chunks
    n_ = n_chunks*n_threads
    return parray_chunks.map(lambda x: x.map(lambda y: y.view(np.ndarray)[:n_])).collect()

def divide_large_parrays_into_chunks(parrays, n_chunks, n_threads, chunk_size):
    """
    Divide a large parray into chunks. Workload is distributed
    by distributing the parrays over the number of workers.
    Returns:
        List of chunks, each chunk contains the worker having indices corresponding
        to the workers having the same indices across all the workers.
    """
    parray_chunks = parrays.map(lambda x: x[np.newaxis]).collect().getSaveRate
    parray_chunks[0] /= n_chunks
